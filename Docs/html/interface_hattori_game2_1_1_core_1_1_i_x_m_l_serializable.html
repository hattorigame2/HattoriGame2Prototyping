Detailed Description

Интрефейс глубокой сериализации объекта в XML

Класс который наследуется от этого интерфейса, а также все вложенные структуры, должны быть помечены аттрибутом Serializable. Сериализуемые поля объекта также должны быть помечены аттрибутом SerializeField

Стандарная реализация интерфейса распологается в статическом классе XMLSerializable

Вот пример объявления класса (с использованием типовых тел), реализующего этот интерфейс

using UnityEngine;
[Serializable]
public class SomeClass : IXMLSerializable<SomeClass>
{
[SerializeField]
public int FirstField;
[SerializeField]
public float SecondField;
[SerializeField]
public OtherClass ThirdField;
public string Serialize()
{
return XMLSerializable.Serialize<SomeClass>(this);
}
public bool TryDeserialize(string xml, out SomeClass deserializedObject)
{
return XMLSerializable.TryDeserialize<SomeClass>(xml, out deserializedObject);
}
public SomeClass Deserialize(string xml)
{
return XMLSerializable.Deserialize<SomeClass>(xml);
}
}
[Serializable]
public class OtherClass
{
[SerializeField]
public string SomeField;
}
